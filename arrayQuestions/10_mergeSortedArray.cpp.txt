// problem : merge sorted array 
// platform: leetcode 88
// difficulty :  easy
// tc =o(n) , sc = o(1)

#include<bits/stdc++.h>
using namespace std ;
class Solution {
public:
    void merge(vector<int>& nums1, int m, vector<int>& nums2, int n) {
        int k  = nums1.size()-1;
        int i = m-1 ;
        int j = n-1;
        while(i>=0 && j>= 0){
            if(nums1[i] < nums2[j]){
                nums1[k--] = nums2[j--];
            }
            else{
                nums1[k--] = nums1[i--];
            }
        }
        // if elements remaining in nums2
            while(j>=0){
                nums1[k--] = nums2[j--];
            }
        
    }
};
