// problem :  missing number
// platform:  leetcode 268
// difficulty :  easy

// tc = o(logn) , sc = o(1)
//aprroach : binary search
#include<bits/stdc++.h>
using namespace std ;
class Solution {
public:
    int missingNumber(vector<int>& nums) {
        sort(nums.begin(),nums.end());

        int n = nums.size();
        int s = 0;
        int e = n-1;
        int mid = s+(e-s)/2;
        int ansIndex = -1;
        // as we are storing and computing ans index is initialized

        while(s<=e) {

            int number = nums[mid];
            int index = mid;
            int diff = number - index ; 
            if (diff == 0){
                // move right
                s = mid + 1 ;
            }
                 
            else if  ( diff == 1) {
                ansIndex = index ;  // store 
                e = mid - 1 ;         // compute move left 
            }
            mid = s+ (e-s)/2;           
        }
        if (ansIndex == -1){
            return n ;
        }
        
        return ansIndex ;  
    }  
};


//aprroach: mathematical insights
// tc: o(n) sc : o(1)
class Solution {
public:
    int missingNumber(vector<int>& nums) {
        int n = nums.size();
        int sum  = n*(n+1)/2;
        int arrSum = 0;
        for(int i =  0;i<n;i++){
            arrSum += nums[i];
        }
        return sum - arrSum;
    }
};
