// problem : maximum sum subarray 
// platform: leetcode 53 
// tc = o(n) , sc = o(1)
class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int ans = INT_MIN ;
        int sum = 0 ;
        for(int i = 0 ; i < nums.size() ; i++){
            sum += nums[i];
            if(ans < sum){
                ans = sum ;
            }
            // kadanes logic if sum < 0 then adding anything would make it more negative so reset sum
            if(sum < 0){
                sum = 0;
            }
        }
        return ans ;
    }
};
