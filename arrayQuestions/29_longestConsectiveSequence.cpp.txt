//problem : maximum consecutive sequence 
// leetcode 128

// approach - sorting + iterative tc = o(nlogn) sc = o(1) - brute force
class Solution {
public:
    int longestConsecutive(vector<int>& nums) {
        if(nums.size()== 0) return 0 ;
        sort(nums.begin(),nums.end());
        int longestConsectiveSeq = 1;
        int currentSeqLength= 1;
        for(int i = 0 ; i<nums.size()-1;i++){
            if(nums[i] == nums[i+1]){
                continue;
            }
            else if(nums[i]+1 == nums[i+1]){
                currentSeqLength ++;
            }
            else{
             longestConsectiveSeq = max(longestConsectiveSeq,currentSeqLength);
              currentSeqLength = 1;    
            }
        }
        return max(longestConsectiveSeq,currentSeqLength);
    }
};

// optimised version - hashSet
class Solution {
public:
    int longestConsecutive(vector<int>& nums) {
      unordered_set<int>hashSet;
        for(int num :nums){
            hashSet.insert(num);
        }
        int longestSeq = 0;
        for(int num : hashSet){
            if(hashSet.find(num-1) == hashSet.end()){
                int currentSeq = 1;
                int currentElement = num;
                while(hashSet.find(currentElement+1) != hashSet.end()){
                    currentElement ++;
                    currentSeq ++;
                }
                longestSeq = max(longestSeq,currentSeq);
            }
        }
        return longestSeq;
    }
};
