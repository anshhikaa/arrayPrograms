// approach inplace ->value encoding
include<bits/stdc++.h>
using namespace std ;
class Solution {
  public:
    // Function to count the frequency of all elements from 1 to N in the array.
    vector<int> frequencyCount(vector<int>& arr) {
        int n = arr.size();
        // 0 based indexing
        for(int i = 0 ;i<n;i++){
            arr[i]--;
        }
        // value encoding
        for(int i = 0 ; i<n;i++){
            int index = arr[i]%n;
            arr[index] += n ;
        }
        //count frequency
        for(int i = 0 ;i<n;i++){
            arr[i]= arr[i]/n;
        }
        return arr;
    }
};
